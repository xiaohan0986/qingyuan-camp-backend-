# 数据库订阅功能升级指南

## 📋 目标
将用户订阅状态从本地存储改为数据库存储，实现跨设备同步。

---

## 🔧 第一步：创建数据库表

在 phpMyAdmin 中执行：

```sql
-- 用户订阅记录表
CREATE TABLE IF NOT EXISTS `user_subscriptions` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` int(11) DEFAULT NULL COMMENT '用户 ID',
  `phone` varchar(20) DEFAULT NULL COMMENT '手机号',
  `subscribe_type` varchar(50) DEFAULT NULL COMMENT '订阅类型：progress_notify',
  `template_id` varchar(100) DEFAULT NULL COMMENT '模板 ID',
  `subscribed` tinyint(1) DEFAULT 0 COMMENT '是否已订阅：0=未订阅，1=已订阅',
  `subscribe_time` datetime DEFAULT NULL COMMENT '订阅时间',
  `last_push_time` datetime DEFAULT NULL COMMENT '最后推送时间',
  `push_count` int(11) DEFAULT 0 COMMENT '推送次数',
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_phone` (`phone`),
  KEY `idx_subscribe_type` (`subscribe_type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户订阅记录表';
```

## 📱 第三步：修改小程序代码

### 修改 `pages/progress/progress.js`

#### 1. 修改 `loadSubscribeConfig` 函数

```javascript
// 加载订阅配置（从数据库读取订阅状态）
loadSubscribeConfig: function() {
  const that = this;
  
  // 1. 先获取订阅配置（模板 ID 等）
  wx.request({
    url: config.getApiUrl('subscribe_config.php'),
    method: 'GET',
    success: function(res) {
      if (res.data && res.data.code === 0) {
        const configData = res.data.data;
        that.setData({
          subscribeTemplateId: configData.template_id || ''
        });
        
        // 2. 从数据库读取订阅状态
        if (that.data.phone && configData.template_id) {
          that.loadSubscriptionFromDB();
        }
      }
    },
    fail: function(err) {
      console.error('[订阅配置] 加载配置失败:', err);
    }
  });
},
```

#### 2. 添加 `loadSubscriptionFromDB` 函数

```javascript
// 从数据库加载订阅状态
loadSubscriptionFromDB: function() {
  const that = this;
  
  wx.request({
    url: config.getApiUrl('get_subscription_status.php'),
    method: 'GET',
    data: {
      phone: that.data.phone
    },
    success: function(res) {
      console.log('[订阅] 数据库订阅状态:', res.data);
      
      if (res.data && res.data.code === 0) {
        const subscribed = res.data.data.subscribed;
        
        that.setData({
          hasSubscribedProgress: subscribed === true
        });
        
        console.log('[订阅] 设置状态:', {
          subscribed: subscribed,
          hasSubscribedProgress: subscribed === true,
          显示按钮：subscribed !== true && that.data.subscribeTemplateId
        });
      }
    },
    fail: function(err) {
      console.error('[订阅] 查询数据库失败:', err);
    }
  });
},
```

#### 3. 修改 `requestSubscribe` 函数

```javascript
// 请求用户订阅（由用户点击触发）
requestSubscribe: function() {
  const that = this;
  
  console.log('[订阅] 用户点击订阅按钮');
  
  if (!this.data.subscribeTemplateId) {
    console.error('[订阅] 未配置订阅模板 ID');
    wx.showToast({
      title: '订阅配置未设置',
      icon: 'none'
    });
    return;
  }
  
  wx.requestSubscribeMessage({
    tmplIds: [this.data.subscribeTemplateId],
    success: function(res) {
      console.log('[订阅] 授权结果:', res);
      
      if (res[that.data.subscribeTemplateId] === 'accept') {
        // 用户同意订阅
        wx.setStorageSync('hasSubscribedProgress', true);  // 本地存储（备用）
        that.setData({ hasSubscribedProgress: true });     // 更新 UI
        
        // 保存到数据库
        that.saveSubscriptionToDB();
        
        wx.showToast({
          title: '订阅成功',
          icon: 'success'
        });
        
        console.log('[订阅] 订阅成功，隐藏按钮');
        
      } else if (res[that.data.subscribeTemplateId] === 'reject') {
        // 用户拒绝订阅
        console.log('[订阅] 用户拒绝，继续显示按钮');
        wx.showToast({
          title: '如需接收通知，请再次点击订阅',
          icon: 'none',
          duration: 2000
        });
      }
    },
    fail: function(err) {
      console.error('[订阅] 授权失败:', err);
      wx.showToast({
        title: '请点击按钮进行订阅',
        icon: 'none'
      });
    }
  });
},
```

---

#### 4. 修改 `saveSubscription` 函数

```javascript
// 保存订阅关系到后台（数据库）
saveSubscriptionToDB: function() {
  const that = this;
  
  wx.request({
    url: config.getApiUrl('save_subscription.php'),
    method: 'POST',
    data: {
      phone: that.data.phone,
      template_id: that.data.subscribeTemplateId,
      subscribed: 1
    },
    success: function(res) {
      console.log('[订阅] 保存到数据库成功:', res);
    },
    fail: function(err) {
      console.error('[订阅] 保存到数据库失败:', err);
    }
  });
},
```

## 📊 数据流向

```
用户进入页面
    ↓
加载订阅配置（subscribe_config.php）
    ↓
从数据库查询订阅状态（get_subscription_status.php）
    ↓
显示/隐藏订阅按钮
    ↓
用户点击订阅
    ↓
微信授权弹窗
    ↓
用户允许
    ↓
保存到数据库（save_subscription.php）
    ↓
隐藏按钮
```
